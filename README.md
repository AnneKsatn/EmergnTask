#БД 

PostgreSQL. Необходимо:

1. Установить PostgreSQL
2. Создать бд "emergn" 
3. Создать таблицу "users"

CREATE TABLE users
(
    Id SERIAL PRIMARY KEY,
    Email CHARACTER VARYING(30),
    Login CHARACTER VARYING(30),
    Password CHARACTER VARYING(30),
	Name CHARACTER VARYING(30)
);
